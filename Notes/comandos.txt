try:
    df = pd.read_csv("/home/ranilton/Área de Trabalho/Data_Analisys/data/SATS_PESQ.csv")

    st.error("ERRO: O arquivo CSV está vazio ou contém apenas cabeçalhos.")
except pd.errors.ParserError:
    st.error("ERRO: O arquivo CSV está mal formatado e o pandas não conseguiu lê-lo.")
    st.write("Tente abrir o arquivo em um editor de texto para verificar sua estrutura.")
except Exception as e:
    st.error(f"Ocorreu um erro inesperado ao carregar o arquivo: {e}")
